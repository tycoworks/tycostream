import type { StreamingAdapter, ParsedEvent } from './streamingAdapter.js';
import type { SourceSchema } from '../core/schema.js';
import { logger, truncateForLog } from '../core/logger.js';
 
/**
 * Materialize-specific streaming adapter.
 * Handles COPY/SUBSCRIBE queries and ENVELOPE UPSERT protocol parsing.
 */
export class MaterializeStreamingAdapter implements StreamingAdapter {
  private log = logger.child({ component: 'materialize-adapter' });
  private columnNames: string[];
 
  constructor(private schema: SourceSchema) {
    // Initialize column names for COPY stream parsing
    // With ENVELOPE UPSERT, output format is: [mz_timestamp, mz_state, key_columns..., value_columns...]
    const keyFields = schema.fields.filter(f => f.name === schema.primaryKeyField);
    const nonKeyFields = schema.fields.filter(f => f.name !== schema.primaryKeyField);
    this.columnNames = ['mz_timestamp', 'mz_state', ...keyFields.map(f => f.name), ...nonKeyFields.map(f => f.name)];
    
    this.log.debug('MaterializeStreamingAdapter initialized', { 
      columnCount: this.columnNames.length,
      columns: this.columnNames,
      primaryKeyField: schema.primaryKeyField
    });
  }
 
  createStreamQuery(): string {
    const keyColumn = this.schema.primaryKeyField;
    const query = `COPY (SUBSCRIBE TO ${this.schema.sourceName} ENVELOPE UPSERT (KEY (${keyColumn})) WITH (SNAPSHOT)) TO STDOUT`;
    this.log.debug('Created streaming query', { query });
    return query;
  }
 
  getColumnOrder(): string[] {
    return this.columnNames;
  }
 
  parseStreamChunk(chunk: Buffer): ParsedEvent[] {
    const events: ParsedEvent[] = [];
    
    try {
      const lines = chunk.toString('utf8').split('\n');
      
      for (const line of lines) {
        if (!line.trim()) continue;
        
        const event = this.parseLine(line);
        if (event) {
          events.push(event);
        }
      }
    } catch (error) {
      this.log.error('Failed to parse chunk', { 
        error,
        chunk: truncateForLog(chunk.toString('utf8'))
      });
    }
    
    return events;
  }
 
  private parseLine(line: string): ParsedEvent | null {
    const values = line.split('\t');
    
    if (values.length !== this.columnNames.length) {
      this.log.warn('Row column count mismatch', { 
        expected: this.columnNames.length, 
        actual: values.length,
        line: truncateForLog(line)
      });
      return null;
    }
    
    // Parse the row
    const row: Record<string, any> = {};
    let mzTimestamp = '0';
    let mzState = 0;
    
    this.columnNames.forEach((name, index) => {
      const value = values[index];
      
      if (name === 'mz_timestamp') {
        mzTimestamp = value || '0';
      } else if (name === 'mz_state') {
        mzState = parseInt(value || '0', 10);
      } else {
        // Find the field definition
        const field = this.schema.fields.find(f => f.name === name);
        const fieldType = field?.type || 'String';
        row[name] = this.parseValue(value || '', fieldType);
      }
    });
    
    // Add timestamp to row for deduplication
    row._timestamp = mzTimestamp;
    
    // Determine event type based on mz_state
    // mz_state: 1 = insert/update, -1 = delete
    const eventType = mzState === -1 ? 'delete' : 'insert';
    
    this.log.debug('Parsed event', {
      eventType,
      primaryKey: row[this.schema.primaryKeyField],
      timestamp: mzTimestamp
    });
    
    return {
      row,
      eventType,
      timestamp: BigInt(mzTimestamp)
    };
  }
 
  private parseValue(value: string, type: string): any {
    if (value === '\\N') return null;
    
    // Handle GraphQL type modifiers
    const baseType = type.replace('!', '').replace('[', '').replace(']', '');
    
    switch (baseType) {
      case 'Int':
        return parseInt(value, 10);
      case 'Float':
        return parseFloat(value);
      case 'Boolean':
        return value === 't' || value === 'true';
      default:
        return value;
    }
  }
}
